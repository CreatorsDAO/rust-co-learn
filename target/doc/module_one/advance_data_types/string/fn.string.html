    // 1 &str
    // 字符串字面量实际上存放在程序的只读数据段中，在程序运行时会被加载到内存中读取
    let s = "Hello Rust";
    let mut s1 = "Hello Go";

    s1 = "Hello Rust";
    println!("{}", s1);

    // 2 String
    // String 通过指针指向存放在堆上的字符串

    let s2 = String::from("Hello Rust");

    // String 有三个word：ptr、len、cap,可以直接通过方法访问

    let cap = s2.capacity();
    let len = s2.len();
    let ptr = s2.as_ptr();

    println!("len {}", len);
    println!("cap {}", cap);
    println!("pointer {:?}", ptr);

    // 3 字符串切片
    // String本质上上一个u8序列，支持切片操作

    let s = String::from("Hello Rust");

    let slice1 = &s[0..5];
    let slice2 = &s[6..10];

    println!("slice1: {}", slice1);
    println!("slice2: {}", slice2);
